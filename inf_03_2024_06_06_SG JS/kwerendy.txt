CREATE DATABASE konferencja;

SELECT imie, nazwisko, koszt FROM uczestnicy WHERE zaplacono=0;

SELECT AVG(koszt) AS 'Średni koszt', SUM(koszt) AS 'Całkowity koszt', COUNT(imie) AS 'Liczba uczestników' FROM uczestnicy;

SELECT nazwa, nazwisko, email FROM uczestnicy, wyklady
WHERE uczestnicy.wyklady_id=wyklady.id AND zaplacono=1;

ALTER TABLE `uczestnicy` DROP `haslo`;
